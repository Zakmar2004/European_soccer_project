SELECT
Match.id,
Country.name AS country_name,
League.name AS league_name,
season,
stage,
match.date,
HT.team_long_name AS home_team,
AT.team_long_name AS away_team,
home_team_goal,
away_team_goal,
B365H,
B365D,
B365A,
BSH,
BSD,
BSA,
BWH,
BWD,
BWA,
GBH,
GBD,
GBA,
IWH,
IWD,
IWA,
LBH,
LBD,
LBA,
PSH,
PSD,
PSA,
SJH,
SJD,
SJA,
VCH,
VCD,
VCA,
WHH,
WHD,
WHA,
--
HTA.buildUpPlaySpeed AS HT_buildUpPlaySpeed,
HTA.buildUpPlayDribbling AS HT_buildUpPlayDribbling,
HTA.buildUpPlayPassing AS HT_buildUpPlayPassing,
HTA.buildUpPlayPositioningClass AS HT_buildUpPlayPositioningClass,
HTA.chanceCreationPassing AS HT_chanceCreationPassing,
HTA.chanceCreationCrossing AS HT_chanceCreationCrossing,
HTA.chanceCreationShooting AS HT_chanceCreationShooting,
HTA.chanceCreationPositioningClass AS
HT_chanceCreationPositioningClass,
HTA.defencePressure AS HT_defencePressure,
HTA.defenceAggression AS HT_defenceAggression,
HTA.defenceTeamWidth AS HT_defenceTeamWidth,
HTA.defenceDefenderLineClass AS HT_defenceDefenderLineClass,
--
ATA.buildUpPlaySpeed AS AT_buildUpPlaySpeed,
ATA.buildUpPlayDribbling AS AT_buildUpPlayDribbling,
ATA.buildUpPlayPassing AS AT_buildUpPlayPassing,
ATA.buildUpPlayPositioningClass AS AT_buildUpPlayPositioningClass,
ATA.chanceCreationPassing AS AT_chanceCreationPassing,
ATA.chanceCreationCrossing AS AT_chanceCreationCrossing,
ATA.chanceCreationShooting AS AT_chanceCreationShooting,
ATA.chanceCreationPositioningClass AS
AT_chanceCreationPositioningClass,
ATA.defencePressure AS AT_defencePressure,
ATA.defenceAggression AS AT_defenceAggression,
ATA.defenceTeamWidth AS AT_defenceTeamWidth,
ATA.defenceDefenderLineClass AS AT_defenceDefenderLineClass
FROM Match
JOIN Country ON Country.id = Match.country_id
JOIN League ON League.id = Match.league_id
LEFT JOIN Team AS HT ON HT.team_api_id = Match.home_team_api_id
LEFT JOIN Team AS AT ON AT.team_api_id = Match.away_team_api_id
LEFT JOIN Team_Attributes AS HTA ON HTA.team_api_id = HT.team_api_id
AND HTA.date = (SELECT MAX(date) FROM Team_Attributes WHERE
team_api_id = HT.team_api_id AND date <= Match.date)
LEFT JOIN Team_Attributes AS ATA ON ATA.team_api_id = AT.team_api_id
AND ATA.date = (SELECT MAX(date) FROM Team_Attributes WHERE
team_api_id = AT.team_api_id AND date <= Match.date)
ORDER BY Match.date
